nasm汇编笔记

nasm计算Fibonacci数列0-200项，效果示意图如下：


思想

使用两个数组表示fib数列的两个数的每一位。

code：

section .text
global main

main:

mov eax,4  
mov ebx,1
mov ecx,string
mov edx,40
int 80h

mov eax, 3
mov ebx, 0
mov ecx, Str
mov edx, 20
int 80h

xor ebx,ebx
xor eax,eax
mov edx,Str
mov eax,0
mov [fiba],eax
mov eax,1
mov [fibb],eax ;初始化两个数组
xor eax,eax

funm:

movzx ecx,byte[edx]
inc edx
cmp ecx,byte '0'
jb getm
cmp ecx,byte '9'
ja getm
lea eax,[eax + eax * 4]
lea eax,[eax * 2 + ecx - '0']
jmp funm   ;得到第一个输入

getm:

mov [m],eax
xor ebx,ebx
xor eax,eax

funn:

movzx ecx,byte[edx]
inc edx
cmp ecx,byte '0'
jb getn
cmp ecx,byte '9'
ja getn
lea eax,[eax + eax * 4]
lea eax,[eax * 2 + ecx - '0']
jmp funn  ;得到第二个输入

getn:

inc eax
mov [n],eax

fib:

mov eax,dword[m]
cmp eax,0
jna fibprint
sub eax,1
mov [m],eax
mov eax,dword[n]
sub eax,1
mov [n],eax
xor r8,r8
xor r9,r9
xor eax,eax
xor ebx,ebx

comp:

cmp r8,100
ja loopend
movzx eax,byte[fibb + r8]
movzx ebx,byte[fiba + r8]
mov [fiba + r8],al
add eax,ebx
add rax,r9
cmp eax,10
jb notsubten
sub eax,10
mov r9,1   ；循环100次，代表100位整数
jmp move

notsubten:

mov r9,0

move:

mov [fibb + r8],al
inc r8
jmp comp

loopend:

jmp fib

fibprint:

mov eax,dword[n]
cmp dword[n],0
jna end
sub eax,1
mov [n],eax
xor ecx,ecx
xor eax,eax
xor edx,edx
xor ebx,ebx
mov eax,fiba

b:

mov ebx,101
add eax,100

c:

cmp byte[eax],0
jne output
dec eax
dec ebx
jmp c

output:

cmp ebx,0
je colorchoose
mov [store1],eax
mov [store2],ebx
mov eax,[eax]
add eax,'0'
mov byte[printbuf],al
mov eax,4
mov ebx,1
mov ecx,printbuf   ;printbuf存当前位的数字，地址向前减，输出每一位
mov edx,1
int 80h
mov eax,dword[store1]
mov ebx,dword[store2]
dec eax
dec ebx
jmp output

colorchoose: ;实现彩色输出

mov eax,[colorcontrol]
cmp eax,1
je printblue
cmp eax,2
je printred
cmp eax,3
je printdefault
cmp eax,4
je printyellow

printblue:

mov eax, 4
mov ebx, 1
mov ecx, color_blue
mov edx, color_blue.len
int 80h
mov eax,[colorcontrol]
add eax,1
mov [colorcontrol],eax
jmp printstr

printred:

mov eax, 4
mov ebx, 1
mov ecx, color_red
mov edx, color_red.len
int 80h
mov eax,[colorcontrol]
add eax,1
mov [colorcontrol],eax
jmp printstr

printdefault:

mov eax, 4
mov ebx, 1
mov ecx, color_default
mov edx, color_default.len
int 80h
mov eax,[colorcontrol]
add eax,1
mov [colorcontrol],eax
jmp printstr

printyellow:

mov eax, 4
mov ebx, 1
mov ecx, color_yellow
mov edx, color_yellow.len
int 80h
mov eax,[colorcontrol]
sub eax,3
mov [colorcontrol],eax    
jmp printstr

printstr:

mov eax,4
mov ebx,1
mov ecx,linebreak
mov edx,1
int 80h


xor eax,eax
xor ebx,ebx

comp1:

cmp r8,100
ja loopend1
movzx eax,byte[fibb + r8]
movzx ebx,byte[fiba + r8]
mov [fiba + r8],al
add eax,ebx
add rax,r9
cmp eax,10
jb notsubten1
sub eax,10
mov r9,1
jmp move1

notsubten1:

mov r9,0

move1:

mov [fibb + r8],al
inc r8
jmp comp1

loopend1:

jmp fibprint

end:

mov eax, 1
mov ebx, 0
int 80h

section .bss
Str:resd 10
printbuf:resd 1
offset:resd 1
printadd:resd 1
fiba:resb 110
fibb:resb 110
store1:resd 1
store2:resd 1
section .data
loopnum:equ 100
m:dd 0
n:dd 0
linebreak:db 10
divnum:dd 10
bufsize:db 1
string:db ‘please imput x and y with a spacebreak: ‘
colorcontrol:dd 1
color_red: db 1Bh, ‘[31;5m’, 0
.len equ $ - color_red
color_blue: db 1Bh, ‘[34;1m’, 0
.len equ $ - color_blue
color_default: db 1Bh, ‘[37;5m’, 0
.len equ $ - color_default
color_yellow: db 1Bh, ‘[33;1m’, 0
.len equ $ - color_yellow

注意事项

  1. mov指令，寄存器向地址mov，mov长度为寄存器长度，地址向寄存器mov，长度可以手动指定，也可以不指定。movzx，高位补0
  2. .len equ $ - add 可以用来代表一个串的长度
  3. div指令，eax，edx必须置0，否则会报浮点数错误
  4. 地址空间申请是连续申请，因此申请100个byte的空间最好多申请10个，防止后续写入破坏原地址内的值
  5. 本次实验没有用到push指令，64位操作系统下的push指令要使用rax，与标准32位nasm略有区别
